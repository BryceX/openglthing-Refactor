#version 410

in vec4 vNormal;

in vec2 vUV;


out vec4 FragColor;

uniform sampler2D diffuseMap;

uniform vec3 lightDirection = vec3(0,-1,0);

uniform vec3 lightColor = vec3(0.2f,1,.4f);

uniform vec3 ambientColor = vec3(1,1,.2f);

void main()

{

vec3 color = texture(diffuseMap,vUV).xyz;
vec3 ambient = ambientColor*color;


float lambert = max(0.0f, dot(vNormal.xyz, -lightDirection));

vec3 diffuse = lightColor*lambert*color;
FragColor = vec4(ambient+diffuse, 1); 

}
