#version 410
layout(location=0) in vec4 Position;
layout(location=1) in vec4 2Normal;
layout(location=2) in vec2 UV;
out vec4 vNormal;
out vec2 vUV;

uniform mat4 Projection;
uniform mat4 Model;
uniform mat4 View;


void main()
{
vNormal = 2Normal;
vUV = UV;
gl_Position = Projection*View*Model*Position;
}
